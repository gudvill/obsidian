Календарь
```python
import calendar
from datetime import datetime, timedelta

now = datetime.now()
year = now.year
month = now.month

cal = calendar.TextCalendar(firstweekday=0)
month_str = cal.formatmonth(year, month)

print(f"# Календарь на {month}.{year}\n")
print("```\n" + month_str + "```")

today = datetime.now()
start_of_week = today - timedelta(days=today.weekday())
end_of_week = start_of_week + timedelta(days=6)

print(f"\n# Календарь недели ({start_of_week.strftime('%d.%m.%Y')} - {end_of_week.strftime('%d.%m.%Y')})\n")

for i in range(7):
    day = start_of_week + timedelta(days=i)
    prefix = "- " if day.date() == today.date() else "  "
    print(f"{prefix}{day.strftime('%A %d.%m.%Y')}")

print(f"\n# План на день: {today.strftime('%A, %d %B %Y')}\n")

for hour in range(8, 22):
    print(f"- {hour:02d}:00 | ")
```
